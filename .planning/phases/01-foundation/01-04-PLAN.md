```typescript
import { type Case } from '@/lib/data/dataTypes';
import { mockCases } from '@/lib/data/mockCases';

// ─── CaseRepository interface (D-21) ─────────────────────────────────────────
// All data access in the app goes through this interface.
// Swap mockCaseRepository for a real API implementation in v2 without touching UI components.

export interface CaseRepository {
  /**
   * Fetch a single case by ID. Rejects with an error if the ID is not found.
   * Downstream: Phase 2 useCaseData hook calls this with the route param caseId.
   */
  fetchCase(id: string): Promise<Case>;

  /**
   * List all available cases (name, entity count, last modified).
   * Downstream: Phase 2 CaseListPage calls this to populate the case list.
   */
  listCases(): Promise<Case[]>;
}

// ─── mockCaseRepository (D-22) ───────────────────────────────────────────────
// The only implementation in Phase 1. Backed by mockCases.ts.
// Components MUST NOT import mockCases.ts directly (D-23).

export const mockCaseRepository: CaseRepository = {
  fetchCase(id: string): Promise<Case> {
    const found = mockCases.find((c) => c.id === id);
    if (!found) {
      return Promise.reject(new Error(`Case not found: ${id}`));
    }
    return Promise.resolve(found);
  },

  listCases(): Promise<Case[]> {
    return Promise.resolve([...mockCases]);
  },
};
```

Key implementation notes:
- `CaseRepository` is a TypeScript interface, not an abstract class — swappable without inheritance overhead
- `mockCaseRepository` uses `Promise.resolve()` / `Promise.reject()` to match the async contract that real API implementations will use
- `listCases()` returns a shallow copy (`[...mockCases]`) to prevent accidental mutation of the mock array
- `fetchCase` returns `Promise.reject` (not `Promise.resolve(undefined)`) for missing IDs — callers must handle the rejection case
