```typescript
import { z } from 'zod';
import { GraphDataSchema } from '@/lib/graph/graphTypes';

// ─── EvidenceFile (D-14) ─────────────────────────────────────────────────────

export const EvidenceFileSchema = z.object({
  id: z.string(),
  name: z.string(),
  categoryId: z.string(),
  type: z.enum(['device_data', 'surveillance', 'financial', 'witness', 'physical']),
  size: z.string(),      // human-readable, e.g. "2.3 MB"
  addedAt: z.string(),   // ISO date string
});

export type EvidenceFile = z.infer<typeof EvidenceFileSchema>;

// ─── EvidenceCategory (D-15) ─────────────────────────────────────────────────

export const EvidenceCategorySchema = z.object({
  id: z.string(),
  name: z.string(),
  files: z.array(EvidenceFileSchema),
});

export type EvidenceCategory = z.infer<typeof EvidenceCategorySchema>;

// ─── Case (D-13) ─────────────────────────────────────────────────────────────

export const CaseSchema = z.object({
  id: z.string(),
  name: z.string(),
  description: z.string(),
  status: z.enum(['active', 'closed', 'archived']),
  createdAt: z.string(),   // ISO date string
  updatedAt: z.string(),   // ISO date string
  graph: GraphDataSchema,
  evidence: z.array(EvidenceCategorySchema),
});

export type Case = z.infer<typeof CaseSchema>;
```

Key notes:
- All schemas use `z.infer` — no manual type duplication (D-16)
- `graph: GraphDataSchema` reuses the shared contract from graphTypes.ts
- `evidence` field added to Case to carry the evidence categories alongside graph data
