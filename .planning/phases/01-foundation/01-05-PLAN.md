```typescript
import {
  getConnectedIds,
  buildGraphFromCase,
  GraphNodeSchema,
  GraphEdgeSchema,
  type GraphEdge,
  type GraphData,
} from '@/lib/graph/graphTypes';

// ─── Fixtures ────────────────────────────────────────────────────────────────

const makeEdge = (overrides: Partial<GraphEdge> & { source: string; target: string }): GraphEdge =>
  GraphEdgeSchema.parse({
    id: 'test-edge',
    label: 'connected to',
    strength: 1.0,
    ...overrides,
  });

const makeNode = (overrides: Record<string, unknown>) =>
  GraphNodeSchema.parse({
    id: 'n-test',
    label: 'Test Node',
    type: 'person',
    status: 'unknown',
    tier: 1,
    parent: null,
    properties: {},
    ...overrides,
  });

const emptyGraph: GraphData = { nodes: [], edges: [] };

// ─── getConnectedIds ──────────────────────────────────────────────────────────

describe('getConnectedIds', () => {
  it('returns IDs of all nodes connected as targets from the given source', () => {
    const edges: GraphEdge[] = [
      makeEdge({ id: 'e1', source: 'A', target: 'B' }),
      makeEdge({ id: 'e2', source: 'A', target: 'C' }),
    ];
    const result = getConnectedIds(edges, 'A');
    expect(result).toEqual(new Set(['B', 'C']));
  });

  it('returns the source ID when the given nodeId appears as a target', () => {
    const edges: GraphEdge[] = [
      makeEdge({ id: 'e1', source: 'A', target: 'B' }),
    ];
    const result = getConnectedIds(edges, 'B');
    expect(result).toEqual(new Set(['A']));
  });

  it('returns empty Set when nodeId has no matching edges', () => {
    const edges: GraphEdge[] = [
      makeEdge({ id: 'e1', source: 'A', target: 'B' }),
    ];
    const result = getConnectedIds(edges, 'X');
    expect(result.size).toBe(0);
  });

  it('returns empty Set when edges array is empty', () => {
    const result = getConnectedIds([], 'A');
    expect(result.size).toBe(0);
  });

  it('returns both source and target neighbors when a node connects in both directions', () => {
    const edges: GraphEdge[] = [
      makeEdge({ id: 'e1', source: 'A', target: 'B' }),
      makeEdge({ id: 'e2', source: 'C', target: 'A' }),
    ];
    const result = getConnectedIds(edges, 'A');
    expect(result).toEqual(new Set(['B', 'C']));
  });

  it('returns a Set (not an array)', () => {
    const result = getConnectedIds([], 'A');
    expect(result).toBeInstanceOf(Set);
  });
});

// ─── buildGraphFromCase ───────────────────────────────────────────────────────

describe('buildGraphFromCase', () => {
  it('returns an object with nodes and edges properties', () => {
    const result = buildGraphFromCase({ graph: emptyGraph });
    expect(result).toHaveProperty('nodes');
    expect(result).toHaveProperty('edges');
  });

  it('returns empty arrays when the case graph has no nodes or edges', () => {
    const result = buildGraphFromCase({ graph: emptyGraph });
    expect(result.nodes).toHaveLength(0);
    expect(result.edges).toHaveLength(0);
  });

  it('returns the same graph data passed in (extraction is identity)', () => {
    const graph: GraphData = {
      nodes: [makeNode({})],
      edges: [makeEdge({ source: 'n-test', target: 'n-other' })],
    };
    const result = buildGraphFromCase({ graph });
    expect(result.nodes).toHaveLength(1);
    expect(result.edges).toHaveLength(1);
  });
});

// ─── GraphNodeSchema (Zod validation) ────────────────────────────────────────

describe('GraphNodeSchema', () => {
  it('parses a valid node without throwing', () => {
    expect(() =>
      makeNode({ type: 'person', status: 'suspect', tier: 0, parent: null })
    ).not.toThrow();
  });

  it('throws ZodError for an invalid EntityType', () => {
    expect(() => makeNode({ type: 'alien' })).toThrow();
  });

  it('throws ZodError for an invalid EntityStatus', () => {
    expect(() => makeNode({ status: 'guilty' })).toThrow();
  });

  it('accepts null parent (root node)', () => {
    expect(() => makeNode({ parent: null })).not.toThrow();
  });

  it('accepts a string parent (child node)', () => {
    expect(() => makeNode({ parent: 'node-001' })).not.toThrow();
  });

  it('accepts all 7 valid EntityType values', () => {
    const types = ['person', 'location', 'event', 'evidence', 'organization', 'vehicle', 'digital'];
    types.forEach((type) => {
      expect(() => makeNode({ type })).not.toThrow();
    });
  });
});

// ─── GraphEdgeSchema (Zod validation) ────────────────────────────────────────

describe('GraphEdgeSchema', () => {
  it('defaults strength to 1.0 when not provided', () => {
    const edge = GraphEdgeSchema.parse({
      id: 'e1',
      source: 'A',
      target: 'B',
      label: 'test',
    });
    expect(edge.strength).toBe(1.0);
  });

  it('preserves explicit strength value', () => {
    const edge = GraphEdgeSchema.parse({
      id: 'e1',
      source: 'A',
      target: 'B',
      label: 'test',
      strength: 0.5,
    });
    expect(edge.strength).toBe(0.5);
  });

  it('throws ZodError when strength exceeds 1.0', () => {
    expect(() =>
      GraphEdgeSchema.parse({ id: 'e1', source: 'A', target: 'B', label: 'test', strength: 2.0 })
    ).toThrow();
  });

  it('throws ZodError when strength is negative', () => {
    expect(() =>
      GraphEdgeSchema.parse({ id: 'e1', source: 'A', target: 'B', label: 'test', strength: -0.1 })
    ).toThrow();
  });
});
```
