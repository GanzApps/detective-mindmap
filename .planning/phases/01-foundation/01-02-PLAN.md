```typescript
import { z } from 'zod';

// ─── EntityType (D-05) ────────────────────────────────────────────────────────

export const EntityTypeSchema = z.enum([
  'person',
  'location',
  'event',
  'evidence',
  'organization',
  'vehicle',
  'digital',
]);

export type EntityType = z.infer<typeof EntityTypeSchema>;

// ─── EntityStatus (D-07) ─────────────────────────────────────────────────────

export const EntityStatusSchema = z.enum([
  'suspect',
  'confirmed',
  'unknown',
]);

export type EntityStatus = z.infer<typeof EntityStatusSchema>;

// ─── GraphNode (D-09, D-10) ──────────────────────────────────────────────────
// NO x/y/z — spatial positions are assigned by renderers at render time.
// tier system: 0=root, 1=branch, 2=leaf — matches 3d_mindmap_fixed.html convention.

export const GraphNodeSchema = z.object({
  id: z.string(),
  label: z.string(),
  type: EntityTypeSchema,
  status: EntityStatusSchema,
  tier: z.number().int().min(0),
  parent: z.string().nullable(),
  properties: z.record(z.string(), z.string()),
});

export type GraphNode = z.infer<typeof GraphNodeSchema>;

// ─── GraphEdge (D-11) ────────────────────────────────────────────────────────

export const GraphEdgeSchema = z.object({
  id: z.string(),
  source: z.string(),
  target: z.string(),
  label: z.string(),
  strength: z.number().min(0).max(1).default(1.0),
});

export type GraphEdge = z.infer<typeof GraphEdgeSchema>;

// ─── GraphData (D-12) ────────────────────────────────────────────────────────

export const GraphDataSchema = z.object({
  nodes: z.array(GraphNodeSchema),
  edges: z.array(GraphEdgeSchema),
});

export type GraphData = z.infer<typeof GraphDataSchema>;

// ─── ENTITY_TYPE_COLOR (D-06) ────────────────────────────────────────────────
// Single source of truth for entity type colors. All renderers import from here.
// Palette derived from the C object in 3d_mindmap_fixed.html, expanded to 7 slots.

export const ENTITY_TYPE_COLOR: Record<EntityType, string> = {
  person:       '#a78bfa',  // violet   — tier-0 color from reference
  organization: '#38bdf8',  // sky blue — tier-1 color from reference
  location:     '#34d399',  // emerald  — tier-3 color from reference
  event:        '#f472b6',  // pink     — tier-2 color from reference
  evidence:     '#fbbf24',  // amber    — new slot
  vehicle:      '#f87171',  // red      — new slot
  digital:      '#a3e635',  // lime     — new slot
};

// ─── Pure utility functions (D-24, D-25, D-26) ───────────────────────────────
// ZERO DOM imports. Tests run with testEnvironment: 'node'.

/**
 * Returns the IDs of all nodes directly connected to nodeId (direct neighbors only,
 * not transitive). Used by both 2D and 3D renderers for node highlighting.
 */
export function getConnectedIds(
  edges: GraphEdge[],
  nodeId: string,
): Set<string> {
  const result = new Set<string>();
  for (const edge of edges) {
    if (edge.source === nodeId) result.add(edge.target);
    if (edge.target === nodeId) result.add(edge.source);
  }
  return result;
}

/**
 * Extracts the GraphData from a Case. Exists as a testable conversion layer
 * even when the conversion is trivial — ensures the boundary is explicit and
 * can be extended without touching call sites.
 */
export function buildGraphFromCase(c: { graph: GraphData }): GraphData {
  return c.graph;
}
```

Key implementation notes:
- `z.enum([...])` produces a union string literal type via `z.infer` — no manual type duplication (per D-16)
- `strength` uses `.default(1.0)` so edges without explicit strength still parse correctly
- `properties: z.record(z.string(), z.string())` matches `Record<string, string>` exactly
- `parent: z.string().nullable()` allows both string IDs and null for root nodes
- `buildGraphFromCase` accepts `{ graph: GraphData }` (structural) rather than importing Case here — Case is defined in dataTypes.ts (Plan 03) which would create a circular dependency
